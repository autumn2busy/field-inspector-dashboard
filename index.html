# Function to clean and standardize website URLs

def _format_website_url(url: str) -> Optional[str]:

   """Clean and format website URL, ensuring it has http/https prefix."""

   if pd.isna(url) or not url:

       return None

     

   url_str: str = str(url).strip()

 

   # If URL doesn't start with http:// or https://, add https://

   if not url_str.startswith(('http://', 'https://')):

       url_str = 'https://' + url_str

 

   # Ensure URL ends with a slash for joining paths

   if not url_str.endswith('/'):

       url_str += '/'

     

   return url_str

# Function to create a careers URL from company website

def _get_careers_url(website: Optional[str], company_name: str) -> str:

   """Create a careers/jobs URL from company website."""

   base_url: Optional[str] = _format_website_url(website) if website else None

 

   if not base_url:

       # Fallback for when we don't have a website

       return f"https://www.google.com/search?q={company_name.replace(' ', '+')}+jobs+careers"

 

   # Try common career page paths (we'll use the first one in the actual carousel)

   career_paths: list = [

       "careers",

       "jobs",

       "join-our-team",

       "employment"

   ]

 

   # In a real implementation, we would check if each path exists

   # For this demo, we'll use the first option

   return f"{base_url}{career_paths[0]}"
